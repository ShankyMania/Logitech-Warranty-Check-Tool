function clearResult(){
  document.getElementById('result').style.display='none';
  document.getElementById('error').style.display='none';

function parseSerial(sn){
  sn=sn.trim().toUpperCase();
  if(sn.length<4) return null;
  const yr2=parseInt(sn.substring(0,2),10);
  const wk=parseInt(sn.substring(2,4),10);
  if(isNaN(yr2)||isNaN(wk)||wk<1||wk>53) return null;
  const year=2000+yr2;
  const jan4=new Date(year,0,4);
  const startOfW1=new Date(jan4);
  startOfW1.setDate(jan4.getDate()-(jan4.getDay()||7)+1);
  const mfgDate=new Date(startOfW1);
  mfgDate.setDate(startOfW1.getDate()+(wk-1)*7);
  if(mfgDate.getFullYear()<2000||mfgDate.getFullYear()>2040) return null;
  return mfgDate;
}

function fmtDate(d){
  return d.toLocaleDateString('en-IN',{year:'numeric',month:'short'});
}

function check(){
  const sn=document.getElementById('sn').value.trim();
  const errEl=document.getElementById('error');
  errEl.style.display='none';

  if(!sn){errEl.textContent='Please enter a serial number.';errEl.style.display='block';return;}
  if(sn.length<6){errEl.textContent='Serial number seems too short. It should be at least 6 characters (e.g. 2239MR1042D9).';errEl.style.display='block';return;}

  const mfgDate=parseSerial(sn);
  if(!mfgDate){errEl.textContent='Could not decode a manufacturing date from this serial number. Make sure to enter it exactly as printed (e.g. 2239MR1042D9).';errEl.style.display='block';return;}

  const graceDays=90;
  const warrantyYears=3;
  const startDate=new Date(mfgDate);
  startDate.setDate(startDate.getDate()+graceDays);
  const endDate=new Date(startDate);
  endDate.setFullYear(endDate.getFullYear()+warrantyYears);

  const today=new Date();
  today.setHours(0,0,0,0);
  const totalMs=endDate-startDate;
  const usedMs=Math.max(0,today-startDate);
  const remainMs=endDate-today;
  const daysLeft=Math.round(remainMs/(1000*60*60*24));
  const pct=Math.min(100,Math.round((usedMs/totalMs)*100));

  const sb=document.getElementById('status-bar');
  const stitle=document.getElementById('status-title');
  const sdesc=document.getElementById('status-desc');
  const fill=document.getElementById('progress-fill');

  sb.className='status-bar';
  fill.className='progress-fill';

  if(daysLeft>180){
    sb.classList.add('status-ok');
    stitle.textContent='Under warranty';
    sdesc.textContent='Your mouse is covered. '+daysLeft.toLocaleString()+' days remaining.';
    fill.classList.add('fill-ok');
  } else if(daysLeft>0){
    sb.classList.add('status-warn');
    stitle.textContent='Warranty expiring soon';
    sdesc.textContent='Only '+daysLeft+' days left — raise a claim soon!';
    fill.classList.add('fill-warn');
  } else {
    sb.classList.add('status-expired');
    stitle.textContent='Warranty expired';
    sdesc.textContent='Expired '+Math.abs(daysLeft)+' days ago ('+fmtDate(endDate)+').';
    fill.classList.add('fill-expired');
  }

  document.getElementById('m-mfg').textContent=fmtDate(mfgDate);
  document.getElementById('m-end').textContent=fmtDate(endDate);
  document.getElementById('m-days').textContent=daysLeft>0?daysLeft.toLocaleString():'Expired';
  document.getElementById('p-pct').textContent=pct+'% used';
  fill.style.width=pct+'%';

  const yr2=sn.substring(0,2);
  const wk=sn.substring(2,4);
  document.getElementById('info-rows').innerHTML=`
    <div class="info-row"><span class="info-key">Serial number</span><span class="info-val" style="font-family:var(--font-mono);font-size:12px">${sn}</span></div>
    <div class="info-row"><span class="info-key">Year decoded</span><span class="info-val">20${yr2}</span></div>
    <div class="info-row"><span class="info-key">Week decoded</span><span class="info-val">Week ${parseInt(wk,10)}</span></div>
    <div class="info-row"><span class="info-key">Manufacturing date</span><span class="info-val">${mfgDate.toLocaleDateString('en-IN',{day:'numeric',month:'long',year:'numeric'})}</span></div>
    <div class="info-row"><span class="info-key">Warranty start (with grace)</span><span class="info-val">${startDate.toLocaleDateString('en-IN',{day:'numeric',month:'long',year:'numeric'})}</span></div>
    <div class="info-row"><span class="info-key">Warranty end</span><span class="info-val">${endDate.toLocaleDateString('en-IN',{day:'numeric',month:'long',year:'numeric'})}</span></div>
    <div class="info-row"><span class="info-key">Warranty type</span><span class="info-val">3-year limited hardware</span></div>
  `;

  document.getElementById('result').style.display='block';
}

document.getElementById('sn').addEventListener('keydown',function(e){if(e.key==='Enter')check();});
